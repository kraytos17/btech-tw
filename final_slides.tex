\section{Introduction}

\begin{frame}{Motivation and Problem Statement}
\textbf{Context}
\begin{itemize}
    \item IoT deployments require cryptography on devices with limited CPU, memory, and energy
    \item Conventional algorithms (AES, RSA, ECC) often exceed microcontroller resource limits
    \item The Mirai botnet attack highlighted risks of insecure embedded systems
\end{itemize}

\vspace{0.3cm}
\textbf{Problem}
\begin{itemize}
    \item Developers sometimes deploy weakened or ad hoc security mechanisms
    \item Lightweight cryptography provides efficient alternatives for constrained platforms
    \item Need to balance security, performance, and implementation cost
\end{itemize}
\end{frame}

\begin{frame}{Objectives \& Contributions}
This study aims to:
\begin{itemize}
    \item Evaluate PRESENT, SPECK, and ASCON on a Raspberry Pi Pico (ARM Cortex-M0+)
    \item Compare execution time, memory usage, and throughput
    \item Identify design trade-offs for constrained embedded environments
\end{itemize}

\vspace{0.2cm}
\textbf{Contributions:}
\begin{itemize}
    \item Reproducible Rust-based benchmarking framework for the RP2040
    \item Empirical data across block cipher, mode, and payload-size dimensions
    \item Practical deployment recommendations based on measured trade-offs
\end{itemize}
\end{frame}

\begin{frame}{Algorithm Overview}
\begin{columns}[t]
\begin{column}{0.33\textwidth}
\centering\textbf{PRESENT}
\begin{itemize}
    \item 64-bit SPN cipher
    \item 80/128-bit keys
    \item 31/32 rounds
    \item $\sim$1{,}570 GE gate count
    \item ISO/IEC 29192-2
\end{itemize}
\end{column}
\begin{column}{0.33\textwidth}
\centering\textbf{SPECK}
\begin{itemize}
    \item ARX Feistel structure
    \item 64--256-bit keys
    \item 27 rounds (64/128)
    \item Fastest in software
    \item No standard
\end{itemize}
\end{column}
\begin{column}{0.33\textwidth}
\centering\textbf{ASCON}
\begin{itemize}
    \item Sponge-duplex AEAD
    \item 128-bit key
    \item 12 rounds ($p^a$, $p^b$)
    \item NIST SP 800-232
    \item AEAD + hashing
\end{itemize}
\end{column}
\end{columns}
\end{frame}

\begin{frame}{Platform \& Evaluation Metrics}
\textbf{Hardware}: Raspberry Pi Pico (ARM Cortex-M0+, 133 MHz, 264 KB SRAM, 2 MB Flash)

\vspace{0.15cm}
\textbf{Software}: Rust \texttt{no\_std}, \texttt{rp2040-hal}, release mode with LTO

\vspace{0.15cm}
\textbf{Metrics:}
\begin{itemize}
    \item \textbf{Execution Time}: hardware timer, interrupts disabled
    \item \textbf{Cycles Per Byte}: $\textit{CPB} = \frac{\textit{total cycles}}{\textit{total bytes}}$
    \item \textbf{Throughput}: Mbps from fixed-size buffer encryption
    \item \textbf{Memory}: ROM (binary size) + RAM (state/buffer analysis)
\end{itemize}

\vspace{0.15cm}
\textbf{Procedure}: 1000+ iterations, warm-up runs, cross-validated with test vectors
\end{frame}

\section{Discussion}

\begin{frame}{Key Takeaways}
\begin{itemize}
    \item \textbf{SPECK}: Best raw throughput (23.67 Mbps) -- optimal for bulk encryption on microcontrollers
    \item \textbf{ASCON}: Competitive throughput (7.56 Mbps) with built-in AEAD -- gap narrows to $\sim$10\% at 256B+ payloads
    \item \textbf{PRESENT}: 70x slower in software but remains relevant for hardware-constrained RFID-class devices
    \item \textbf{Mode selection matters}: CTR offers semantic security with only 11\% throughput overhead vs ECB
\end{itemize}
\end{frame}

\begin{frame}{Use-Case Recommendations}
\begin{enumerate}
    \item \textbf{Real-time sensor data}: SPECK-64/128 in CTR mode
    \begin{itemize}
        \item 23.67 Mbps, 32 B RAM, for high-bandwidth sensor streams
    \end{itemize}
    \item \textbf{Secure IoT with auth}: ASCON
    \begin{itemize}
        \item 7.56 Mbps, AEAD, NIST standardized
    \end{itemize}
    \item \textbf{RFID tags}: PRESENT
    \begin{itemize}
        \item Ultra-low gate count, adequate for low-bandwidth comms
    \end{itemize}
    \item \textbf{Firmware authentication}: ASCON-hash
    \begin{itemize}
        \item 320-bit state, 12-round permutation for anti-tampering
    \end{itemize}
    \item \textbf{Hybrid deployment}: SPECK + ASCON
    \begin{itemize}
        \item SPECK for bulk data, ASCON for auth tags
    \end{itemize}
    \item \textbf{Legacy integration}: PRESENT-128 in CBC mode
    \begin{itemize}
        \item Backward compatibility with 128-bit security
    \end{itemize}
\end{enumerate}
\end{frame}
