\section{Introduction}

\begin{frame}{Motivation and Problem Statement}
\textbf{Historical Context}
\begin{itemize}
    \item DES (1977) $\rightarrow$ AES (2001): Evolved for desktop/server computing
    \item IoT devices: constrained CPU, memory, power makes traditional algorithms impractical
    \item Mirai botnet (2016): Exploited insecure embedded devices, with millions hijacked
\end{itemize}

\vspace{0.2cm}
\textbf{Problem}
\begin{itemize}
    \item Developers deploy weakened or ad hoc security
    \item Lightweight cryptography: efficient primitives for constrained platforms
    \item Need to balance security, performance, and implementation cost
\end{itemize}
\end{frame}

\begin{frame}{Objectives \& Contributions}
This study aims to:
\begin{itemize}
    \item Evaluate PRESENT, SPECK, and ASCON on Raspberry Pi Pico (ARM Cortex-M0+)
    \item Compare execution time, memory usage, and throughput
    \item Identify design trade-offs for constrained embedded environments
\end{itemize}

\vspace{0.2cm}
\textbf{Contributions:}
\begin{itemize}
    \item Reproducible Rust-based benchmarking framework for RP2040
    \item Empirical data across block cipher, mode, and payload-size dimensions
    \item Practical deployment recommendations based on measured trade-offs
\end{itemize}
\end{frame}

\vspace{0.15cm}
\textbf{\structure{Key Differentiators:}}
\begin{itemize}
    \item \textbf{\textcolor{darkred}{PRESENT}}: $\sim$1,570 GE (\textbf{hardware-optimized})
    \item \textbf{\textcolor{darkgreen}{SPECK}}: \textbf{42 CPB} (\textbf{software fastest})
    \item \textbf{\textcolor{darkblue}{ASCON}}: \textbf{AEAD} + hashing built-in (\alert{NIST standard})

\begin{frame}{Platform \& Metrics}
\textbf{Hardware}: Raspberry Pi Pico (ARM Cortex-M0+, 133 MHz, 264 KB SRAM, 2 MB Flash)

\vspace{0.15cm}
\textbf{Metrics}:
\begin{itemize}
    \item \textbf{Performance}: Throughput (Mbps), Cycles/Byte (CPB), latency
    \item \textbf{Memory}: ROM footprint (KB), RAM usage (bytes)
    \item \textbf{Energy}: µJ per byte using V-I-t model (I = 80--90 mA per RP2040 datasheet; estimates are relative for comparison)
\end{itemize}

\vspace{0.15cm}
\textbf{Testing Procedure}:
\begin{itemize}
    \item 1000+ iterations per cipher-mode-payload combination
    \item Hardware timer (RP2040 internal counter @ 133 MHz)
    \item Test vector validation (NIST vectors for PRESENT, Beaulieu for SPECK)
    \item Payload sizes: 8B, 16B, 64B, 256B, 1024B
\end{itemize}
\end{frame}


\section{Results}

\begin{frame}{Block Cipher Benchmarks}
\vspace{-0.3cm}
\begin{table}
    \centering
    \begin{tabular}{lrr}
        \toprule
        \textbf{Algorithm} & \textbf{CPB} & \textbf{Mbps} \\
        \midrule
        PRESENT-80 & $2{,}958$ & 0.34 \\
        PRESENT-128 & $3{,}068$ & 0.33 \\
        SPECK-64/96 & 42 & 23.81 \\
        SPECK-64/128 & 42 & 23.67 \\
        ASCON-128 & 132 & 7.56 \\
        \bottomrule
    \end{tabular}
\end{table}
\vspace{0.05cm}
\textbf{\large Key Finding}: \alert{SPECK is \textbf{70x faster}} than PRESENT and \textbf{3x faster} than ASCON (\alert{23.67 Mbps} vs 0.34 Mbps vs 7.56 Mbps).

\vspace{0.15cm}
\small \textit{\structure{Insight}}: Single-block throughput shows the dramatic performance gap: SPECK's ARX design dominates on software platforms.
\end{frame}

\begin{frame}{Mode Impact: ECB vs CBC vs CTR}
\vspace{-0.3cm}
\begin{table}
    \centering
    \small
    \begin{tabular}{lccc}
        \toprule
        \textbf{Cipher Mode} & \textbf{64B (Mbps)} & \textbf{256B (Mbps)} & \textbf{Trade-off} \\
        \midrule
        SPECK-ECB & \textbf{19.04} & \textbf{20.33} & \textbf{\textcolor{darkgreen}{Fastest}}; no IV overhead \\
        SPECK-CBC & 16.07 & 18.90 & $-$16\%; sequential dependency \\
        SPECK-CTR & 17.14 & 19.42 & $-$11\%; \alert{parallelizable, semantic security} \\
        \midrule
        PRESENT-ECB & 0.34 & 0.34 & Fastest; block-aligned \\
        PRESENT-CTR & 0.33 & 0.33 & $\sim$0\% overhead (negligible vs 2,958 CPB) \\
        \bottomrule
    \end{tabular}
\end{table}

\vspace{0.2cm}
\textbf{\structure{Key Insight}}: Mode overhead is \textbf{\alert{cipher-dependent}}. SPECK loses 11–16\% in CBC/CTR; PRESENT's overhead is \textbf{\structure{negligible}} due to dominant S-box cost.

\vspace{0.1cm}
\textbf{\textcolor{darkgreen}{Recommendation}}: Use \textbf{CTR} for applications needing semantic security and parallelism without sacrificing performance.
\end{frame}

\begin{frame}{Why This Performance Gap?}
\textbf{\structure{Architectural Explanation:}}
\begin{itemize}
    \item \textbf{\textcolor{darkgreen}{SPECK (ARX)}}: Add, Rotate, XOR $\rightarrow$ \alert{\textbf{single-cycle ARM}} instructions (\texttt{ADDS}, \texttt{RORS}, \texttt{EORS}), no memory lookups
    \item \textbf{\textcolor{darkred}{PRESENT (SPN)}}: Bit-permutation requires \alert{\textbf{64 separate}} bit extractions/reinsertions per round, 31/32 rounds = \textbf{70x overhead}
    \item \textbf{\textcolor{darkblue}{ASCON (Sponge)}}: 64-bit words align with Cortex-M0+ datapath, but 12-round $p^a$ permutation \textbf{heavier} than SPECK's 5-instruction round
\end{itemize}

\vspace{0.2cm}
\textbf{\structure{Bottom Line}}: \alert{\textbf{Hardware-optimized}} ciphers (PRESENT) \textbf{sacrifice software}; \textbf{software-optimized} ciphers (SPECK) \textbf{exploit} architectural fit.
\end{frame}

\vspace{0.3cm}
\small \structure{\textbf{Summary}}: \textcolor{darkgreen}{SPECK} dominates \textbf{performance} and \textbf{ROM}; \textcolor{darkblue}{ASCON} provides \textbf{standardization and AEAD}; \textcolor{darkred}{PRESENT} excels in \textbf{hardware} but \textbf{poor software}.
\end{frame}

\begin{frame}{Security Considerations}
\begin{itemize}
    \item \textbf{\textcolor{darkblue}{ASCON}}: \alert{\textbf{NIST SP 800-232}}, proven indifferentiability bounds, no attacks on full 12-round permutation \textbf{strongest}
    \item \textbf{\textcolor{darkred}{PRESENT}}: ISO/IEC 29192-2, \textbf{provable SPN} bounds (max diff probability $2^{-2}$), \alert{confidentiality only} (no authentication)
    \item \textbf{\textcolor{darkgreen}{SPECK}}: \alert{\textbf{No provable}} bounds, best attacks reduce 20/27 rounds; \alert{ISO rejection} (NSA origin, geopolitical objections) (\textit{research use only})
    \item \textbf{\structure{Side-channels}}: \alert{\textbf{SPECK}} vulnerable to CPA (50 traces on 8-bit); \textbf{PRESENT's} constant-time ops more resistant; \textbf{ASCON} shows moderate resistance
\end{itemize}

\vspace{0.2cm}
\small \textit{\structure{Trade-off}}: \textcolor{darkblue}{ASCON} offers strongest security; \textcolor{darkred}{PRESENT} offers provable bounds; \textcolor{darkgreen}{SPECK} is fastest but has weakest guarantees.
\end{frame}

\begin{frame}{Key Takeaways: Design Lessons}
\textbf{\structure{Algorithm Design Trade-offs:}}
\begin{itemize}
    \item \alert{\textbf{Hardware $\neq$ Software}}: PRESENT's bit-permutation is elegant in silicon (1,570 GE) but catastrophic in code (2,958 CPB)
    \item \textbf{\structure{Instruction Set Alignment}}: SPECK's ARX operations map \textbf{perfectly to ARM} instructions (42 CPB); sponge-based designs add fixed overhead (427 cycles init)
    \item \textbf{\structure{AEAD vs Raw Cipher}}: ASCON's built-in authentication justifies overhead; SPECK+HMAC would be bulkier
\end{itemize}

\vspace{0.2cm}
\textbf{\structure{For Practitioners:}}
\begin{itemize}
    \item Choose by \alert{\textbf{workload}}, not by raw speed: \textcolor{darkgreen}{\textbf{sensor data}} (SPECK) versus \textcolor{darkblue}{\textbf{authenticated IoT}} (ASCON) versus \textcolor{darkred}{\textbf{legacy}} (PRESENT)
    \item \textbf{\structure{Standardization matters}}: NIST-endorsed ASCON gaining adoption; SPECK rejected by ISO (politics and no provable bounds)
    \item \textbf{\structure{Context is everything}}: RP2040 (Cortex-M0+) results don't generalize to hardware, 8-bit MCUs, or server environments
\end{itemize}
\end{frame}


\section{Recommendations}

\begin{frame}{Use-Case Recommendations}
\begin{enumerate}
    \item \textbf{Real-time sensor data}: SPECK-64/128 (CTR), 23.67 Mbps, 32B RAM
    \item \textbf{Secure IoT with auth}: ASCON, 7.56 Mbps, AEAD, NIST standardized
    \item \textbf{RFID tags}: PRESENT, ultra-low gate count (~1570 GE)
    \item \textbf{Firmware authentication}: ASCON-hash, 320-bit state, 12-round $p^a$
    \item \textbf{Hybrid deployment}: SPECK (bulk) + ASCON (auth tags)
    \item \textbf{Legacy integration}: PRESENT-128 (CBC) for backward compatibility
\end{enumerate}
\end{frame}

\begin{frame}{Limitations \& Future Work}
\textbf{\structure{Current Limitations:}}
\begin{itemize}
    \item \textbf{Single platform}: RP2040 only, architecture-specific results
    \item \textbf{Single language}: Rust implementation, no C/C++ comparison
    \item \textbf{No side-channel analysis}: Timing and power attacks not evaluated
\end{itemize}

\vspace{0.2cm}
\textbf{\structure{Natural Extensions of This Work:}}
\begin{enumerate}
    \item Port implementations to ESP32 and compare against RP2040 results
    \item Implement SPECK in C with compiler optimizations and benchmark against Rust
    \item Conduct comparative energy analysis using actual multimeter measurements
    \item Expand to test multiple RP2040 clock speeds (50 MHz, 100 MHz, 133 MHz)
    \item Compare ASCON-128 throughput against ASCON-128a on same platform
\end{enumerate}
\end{frame}
